find_package(OpenMP REQUIRED)
if(OPENMP_FOUND)
  set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${OpenMP_C_FLAGS}")
  set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS}")
else(OPENMP_FOUND)
  message (STATUS "Not found OpenMP")
endif()

find_package(GTest REQUIRED)

add_executable(${PROJECT_NAME}_box_sphere_unit collision_box_sphere_unit.cpp)
target_link_libraries(${PROJECT_NAME}_box_sphere_unit ${GTEST_BOTH_LIBRARIES} ${PROJECT_NAME}_bullet ${PROJECT_NAME}_fcl tesseract_geometry ${BULLET_LIBRARIES} ${Boost_LIBRARIES} ${OCTOMAP_LIBRARIES} ${LIBFCL_INCLUDE_DIRS} ${console_bridge_LIBRARIES})
target_compile_options(${PROJECT_NAME}_box_sphere_unit PRIVATE -Wsuggest-override -Wconversion -Wsign-conversion)
target_compile_features(${PROJECT_NAME}_box_sphere_unit PRIVATE cxx_std_11)
target_compile_definitions(${PROJECT_NAME}_box_sphere_unit PRIVATE DATA_DIR="${CMAKE_SOURCE_DIR}/test")
target_include_directories(${PROJECT_NAME}_box_sphere_unit PRIVATE ${GTEST_INCLUDE_DIRS})
add_test(${PROJECT_NAME}_box_sphere_unit ${PROJECT_NAME}_box_sphere_unit)
