find_package(GTest REQUIRED)
find_package(roslib REQUIRED)

add_executable(${PROJECT_NAME}_kdl_unit kdl_kinematics_unit.cpp)
target_link_libraries(${PROJECT_NAME}_kdl_unit ${GTEST_BOTH_LIBRARIES} ${PROJECT_NAME}_kdl tesseract_scene_graph tesseract_geometry Eigen3::Eigen ${roslib_LIBRARIES} ${orocos_kdl_LIBRARIES} ${console_bridge_LIBRARIES})
target_compile_options(${PROJECT_NAME}_kdl_unit PRIVATE -Wsuggest-override -Wconversion -Wsign-conversion)
target_compile_features(${PROJECT_NAME}_kdl_unit INTERFACE cxx_std_11)
target_compile_definitions(${PROJECT_NAME}_kdl_unit PRIVATE DATA_DIR="${CMAKE_SOURCE_DIR}/test")
target_include_directories(${PROJECT_NAME}_kdl_unit PRIVATE ${GTEST_INCLUDE_DIRS} ${roslib_INCLUDE_DIRS})
add_test(${PROJECT_NAME}_kdl_unit ${PROJECT_NAME}_kdl_unit)
